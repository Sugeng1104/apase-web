} // LOGIN if(action === "login"){ return login(e); } // GET USERS if(action === "getUsers"){ return getUsers(); } // GET PROFIL SEKOLAH if(action === "getProfilSekolah"){ return getProfilSekolah(); } // GET KEPSEK if(action === "getKepsek"){ return getKepsek(); } // CEK STATUS FOLDER PB if(action === "checkFolderPB"){ return checkFolderPB(e); } return response({ status:"error", message:"Action tidak ditemukan" }); } catch(error){ return response({ status:"error", message:error.toString() }); } } // ===================================== // doPost // ===================================== function doPost(e){ try{ const data = JSON.parse( e.postData.contents ); const action = data.action; // SAVE PROFIL if(action === "saveProfilSekolah"){ return saveProfilSekolah(data); } // SAVE AKUN if(action === "saveAkun"){ return saveAkun(data); } // UPLOAD FILE ARSIP if(action === "uploadFileArsip"){ return uploadFileArsip(data); } // CREATE FOLDER PB if(action === "createFolderPB"){ return createFolderPB(data); } } // ===================================== // LOGIN // ===================================== function login(e){ const username = String( e.parameter.username || "" ).trim(); const password = String( e.parameter.password || "" ).trim(); const ss = SpreadsheetApp.openById( ACCOUNT_SPREADSHEET_ID ); const sheet = ss.getSheetByName( ACCOUNT_SHEET ); const values = sheet.getDataRange() .getValues(); for(let i = 1; i < values.length; i++){ if( values[i][0] == username && values[i][1] == password ){ return response({ status:"success", username: values[i][0], email: values[i][2], nama_pengguna: values[i][3] }); } } return response({ status:"error", message:"Login gagal" }); } // ===================================== // GET USERS // ===================================== function getUsers(){ const ss = SpreadsheetApp.openById( ACCOUNT_SPREADSHEET_ID ); const sheet = ss.getSheetByName( ACCOUNT_SHEET ); const values = sheet.getDataRange() .getValues(); let data = []; for(let i = 1; i < values.length; i++){ data.push({ username: values[i][0], password: values[i][1], email: values[i][2], nama: values[i][3] }); } return response({ status:"success", data:data }); } // ===================================== // SAVE AKUN // ===================================== function saveAkun(data){ const ss = SpreadsheetApp.openById( ACCOUNT_SPREADSHEET_ID ); const sheet = ss.getSheetByName( ACCOUNT_SHEET ); sheet.appendRow([ data.username, data.password, data.email, data.nama_pengguna ]); return response({ status:"success", message:"Akun berhasil dibuat" }); } // ===================================== // SAVE PROFIL SEKOLAH // ===================================== function saveProfilSekolah(data){ const ss = SpreadsheetApp.openById( DATABASE_SPREADSHEET_ID ); // DATA SEKOLAH const sekolahSheet = ss.getSheetByName( DATA_SEKOLAH_SHEET ); sekolahSheet.appendRow([ new Date(), data.nama_sekolah, data.npsn, data.alamat, data.desa_sekolah, data.kecamatan, data.kabupaten, data.provinsi, data.email ]); // DATA KEPSEK const kepsekSheet = ss.getSheetByName( DATA_KEPSEK_SHEET ); kepsekSheet.appendRow([ data.nama_kepsek, data.nip, data.pangkat, data.alamat_kepsek, data.email_kepsek, data.nohp ]); return response({ status:"success", message:"Profil berhasil disimpan" }); } // ===================================== // GET PROFIL SEKOLAH // ===================================== function getProfilSekolah(){ const ss = SpreadsheetApp.openById( DATABASE_SPREADSHEET_ID ); const sheet = ss.getSheetByName( DATA_SEKOLAH_SHEET ); } // ===================================== // GET DATA KEPSEK // ===================================== function getKepsek(){ const ss = SpreadsheetApp.openById( DATABASE_SPREADSHEET_ID ); const sheet = ss.getSheetByName( DATA_KEPSEK_SHEET ); const values = sheet.getDataRange() .getValues(); const row = values[values.length - 1]; return response({ status:"success", data:{ nama: row[0] || "", nip: row[1] || "", pangkat: row[2] || "", alamat: row[3] || "", email: row[4] || "", nohp: row[5] || "" } }); } // ===================================== // UPLOAD FILE ARSIP // ===================================== function uploadFileArsip(data){ try{ const folder = DriveApp.getFolderById( ARSIP_FOLDER_ID ); const base64 = data.base64.replace( /^data:.+;base64,/, "" ); const bytes = Utilities.base64Decode( base64 ); const blob = Utilities.newBlob( bytes, data.mimeType, data.fileName ); const file = folder.createFile(blob); return response({ status:"success", message: "File berhasil diupload", file_url: file.getUrl() }); } catch(error){ return response({ status:"error", message:error.toString() }); } } // ===================================== // CHECK FOLDER PB // ===================================== function checkFolderPB(e){ try{ const namaPengguna = e.parameter.nama_pengguna; const parentFolder = DriveApp.getFolderById( PB_FOLDER_ID ); const folders = parentFolder.getFoldersByName( namaPengguna ); if(folders.hasNext()){ const folder = folders.next(); return response({ status:"success", exists:true, folderId: folder.getId(), folderUrl: folder.getUrl() }); } return response({ status:"success", exists:false }); } catch(error){ return response({ status:"error", message:error.toString() }); } } // ===================================== // CREATE FOLDER PB // ===================================== function createFolderPB(data){ try{ const parentFolder = DriveApp.getFolderById( PB_FOLDER_ID ); const namaPengguna = data.nama_pengguna; const folders = parentFolder.getFoldersByName( namaPengguna ); let folder; if(folders.hasNext()){ folder = folders.next(); } else{ folder = parentFolder.createFolder( namaPengguna ); } return response({ status:"success", folderId: folder.getId(), folderName: folder.getName(), folderUrl: folder.getUrl(), message: "Folder berhasil dibuat" }); } catch(error){ return response({ status:"error", message:error.toString() }); } } // ===================================== // UPLOAD BUKU PB // ===================================== function uploadBukuPB(data){ try{ const folder = DriveApp.getFolderById( data.folder_id ); const base64 = data.base64.replace( /^data:.+;base64,/, "" ); const bytes = Utilities.base64Decode( base64 ); const blob = Utilities.newBlob( bytes, data.mimeType, data.fileName ); const file = folder.createFile(blob); const ss = SpreadsheetApp.openById( DATABASE_SPREADSHEET_ID ); let sheet = ss.getSheetByName("pb"); // JIKA SHEET BELUM ADA if(!sheet){ sheet = ss.insertSheet("pb"); sheet.appendRow([ "Nama Pengguna", "Nama Buku", "Folder Buku", "Link Buku", "Status" ]); } // SIMPAN DATA sheet.appendRow([ data.nama_pengguna, data.nama_buku, folder.getUrl(), file.getUrl(), "Sudah" ]); // WARNA HIJAU STATUS const lastRow = sheet.getLastRow(); sheet .getRange(lastRow,5) .setBackground("#16a34a") .setFontColor("#ffffff") .setFontWeight("bold"); return response({ status:"success", message: "File berhasil diupload", fileUrl: file.getUrl() }); } catch(error){ return response({ status:"error", message:error.toString() }); } } // ===================================== // RESPONSE JSON // ===================================== function response(data){ return ContentService .createTextOutput( JSON.stringify(data) ) .setMimeType( ContentService .MimeType.JSON );
